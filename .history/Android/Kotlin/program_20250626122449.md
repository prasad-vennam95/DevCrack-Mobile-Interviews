# Prime Number Check

```kotlin

fun isPrime(n: Int): Boolean {
    if (n <= 1) return false
    for (i in 2..Math.sqrt(n.toDouble()).toInt()) {
        if (n % i == 0) return false
    }
    return true
}

fun main() {
    val number = 29
    println("$number is prime: ${isPrime(number)}")
}

// Output: 29 is prime: true
```

## sum of Digits in given  array  should be equal to given number

```kotlin
fun hasArrayTwoCandidates(arr: IntArray, target: Int): Boolean {
    val numMap = mutableSetOf<Int>()
    for (i in 0 until arr.size) {
        val complement = target - arr[i]
        if (numMap.contains(complement)) {
            return true
        }
        numMap.add(arr[i])
    }
    return false
}

fun main() {
    val arr = intArrayOf(1, 4, 45, 6, 10, 8)
    val target = 16
    println("Array has two elements with sum $target: ${hasArrayTwoCandidates(arr, target)}")
}

// Output: Array has two elements with sum 16: true
```

## Reverse a String without using built-in functions

```kotlin
fun reverseString(str: String): String {
    var reversed = ""
    for (i in str.length - 1 downTo 0) {
        reversed += str[i]
    }
    return reversed
}

fun main() {
    val original = "Hello, World!"
    val reversed = reverseString(original)
    println("Original: $original")
    println("Reversed: $reversed")
}

// Output: Original: Hello, World!
//         Reversed: !dlroW ,olleH

```

## Check if a String is a Palindrome

```kotlin

fun isPalindrome(str: String): Boolean {
    val cleanedStr = str.replace(Regex("[^A-Za-z0-9]"), "").lowercase()
    return cleanedStr == cleanedStr.reversed()
}

fun main() {
    val testString = "amma"
    println("$testString is a palindrome: ${isPalindrome(testString)}")
}
// Output: amma is a palindrome: true
```
